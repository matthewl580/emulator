Getting Started

Welcome to ChipCode! This game engine lets you create retro-style games with a 64x64 pixel canvas and built-in audio support.

Quick Start

  1. Open the Editor
  2. Create initialization code for your game state
  3. Write update code for game logic and rendering
  4. Test your game in the live preview
  5. Share your game with others

Core Features

  • 64x64 pixel canvas with automatic scaling
  • Advanced audio synthesis with Tone.js
  • Built-in keyboard input handling
  • Simple collision detection
  • Support for game state and animations

Game Structure

Game Components

Every ChipCode game consists of two main parts:

1. Initialization Code

This runs once when your game starts. Use it to:

  • Set up initial game state
  • Create game objects
  • Initialize audio
  • Set up event handlers

2. Update Code

This runs every frame. Use it to:

  • Handle input
  • Update game state
  • Draw graphics
  • Check collisions
  • Play sounds

Creating Animations

Add simple animations to your game:

// In initCode:
window._animations = {
    explosion: {
        frames: [0, 1, 2, 3, 4],
        current: 0,
        timer: 0
    }
};

// In updateCode:
function updateAnimation(anim) {
    anim.timer++;
    if (anim.timer > 5) {
        anim.timer = 0;
        anim.current = (anim.current + 1) % anim.frames.length;
    }
}

// Draw animation frame
function drawExplosion(x, y) {
    const anim = window._animations.explosion;
    updateAnimation(anim);
    
    // Draw different patterns based on frame
    switch(anim.frames[anim.current]) {
        case 0: drawSmall(x, y); break;
        case 1: drawMedium(x, y); break;
        case 2: drawLarge(x, y); break;
        case 3: drawMedium(x, y); break;
        case 4: drawSmall(x, y); break;
    }
}

Example Game

Coin Collection Game

A complete example game where you collect coins while avoiding enemies. Use the arrow keys to move, collect yellow coins to increase your score, and avoid red enemies!

Initialization Code

// Game state setup
window._state = {
    player: { x: 32, y: 32, score: 0 },
    coins: Array(5).fill().map(() => ({
        x: Math.floor(Math.random() * 64),
        y: Math.floor(Math.random() * 64)
    })),
    enemies: Array(3).fill().map(() => ({
        x: Math.floor(Math.random() * 64),
        y: Math.floor(Math.random() * 64),
        dx: Math.random() > 0.5 ? 1 : -1,
        dy: Math.random() > 0.5 ? 1 : -1
    })),
    gameOver: false
};

// Audio setup
try {
    window._synth = new Tone.PolySynth().toDestination();
    window._bassLine = new Tone.Synth({
        oscillator: { type: 'triangle' },
        envelope: { attack: 0.1, decay: 0.2, sustain: 0.5, release: 0.8 }
    }).toDestination();
    
    // Music patterns
    const melody = ['C4','E4','G4','C5','G4','E4'];
    const bassline = ['C3', 'G2', 'A2', 'F2'];
    
    let mi = 0, bi = 0;
    
    // Setup music loops
    window._melodyLoop = new Tone.Loop((time) => {
        window._synth.triggerAttackRelease(
            melody[mi % melody.length], '8n', time
        );
        mi++;
    }, '4n');
    
    window._bassLoop = new Tone.Loop((time) => {
        window._bassLine.triggerAttackRelease(
            bassline[bi % bassline.length], '2n', time
        );
        bi++;
    }, '2n');
    
    // Sound effects
    window._effects = {
        coinSound: () => window._synth.triggerAttackRelease(['C6', 'E6'], '16n'),
        hitSound: () => window._synth.triggerAttackRelease(['C3', 'G2'], '8n'),
        gameOverSound: () => {
            window._synth.triggerAttackRelease(['C4', 'G3', 'E3', 'C3'], '2n');
            window._melodyLoop.stop();
            window._bassLoop.stop();
        }
    };

    Tone.Transport.bpm.value = 120;
    window._melodyLoop.start(0);
    window._bassLoop.start(0);
    
} catch (e) { 
    console.warn('Audio unavailable', e);
}

Update Code

// Game update loop
(function(){
    const canvas = document.getElementById('game-canvas');
    const ctx = canvas.getContext('2d');
    const state = window._state;
    if (!state || state.gameOver) return;

    // Input handling
    document.onkeydown = (e) => {
        if (state.gameOver) return;
        switch(e.key) {
            case 'ArrowLeft': 
                state.player.x = Math.max(0, state.player.x - 1);
                break;
            case 'ArrowRight':
                state.player.x = Math.min(63, state.player.x + 1);
                break;
            case 'ArrowUp':
                state.player.y = Math.max(0, state.player.y - 1);
                break;
            case 'ArrowDown':
                state.player.y = Math.min(63, state.player.y + 1);
                break;
        }
    };

    // Clear screen and set scale
    ctx.fillStyle = '#000022';
    ctx.fillRect(0, 0, canvas.width, canvas.height);
    const scale = canvas.width / 64;

    // Draw animated background stars
    for (let i = 0; i < 20; i++) {
        ctx.fillStyle = '#ffffff';
        ctx.fillRect(
            ((Math.sin(Date.now()/1000 + i) * 32 + 32) * scale),
            ((Math.cos(Date.now()/1000 + i) * 32 + 32) * scale),
            scale/2,
            scale/2
        );
    }

    // Update and draw coins
    state.coins.forEach(coin => {
        // Collect coins
        if (Math.abs(coin.x - state.player.x) < 2 && 
            Math.abs(coin.y - state.player.y) < 2) {
            state.player.score += 10;
            coin.x = Math.floor(Math.random() * 64);
            coin.y = Math.floor(Math.random() * 64);
            if (window._effects) window._effects.coinSound();
        }

        // Draw coin
        ctx.fillStyle = '#ffff00';
        ctx.fillRect(coin.x * scale, coin.y * scale, scale, scale);
    });

    // Update and draw enemies
    state.enemies.forEach(enemy => {
        // Move enemies
        enemy.x += enemy.dx;
        enemy.y += enemy.dy;
        
        // Bounce off walls
        if (enemy.x <= 0 || enemy.x >= 63) enemy.dx *= -1;
        if (enemy.y <= 0 || enemy.y >= 63) enemy.dy *= -1;

        // Check collision with player
        if (Math.abs(enemy.x - state.player.x) < 2 && 
            Math.abs(enemy.y - state.player.y) < 2) {
            state.gameOver = true;
            if (window._effects) window._effects.gameOverSound();
        }

        // Draw enemy
        ctx.fillStyle = '#ff0000';
        ctx.fillRect(enemy.x * scale, enemy.y * scale, scale, scale);
    });

    // Draw player
    ctx.fillStyle = '#00ff00';
    ctx.fillRect(state.player.x * scale, state.player.y * scale, 
                scale, scale);

    // Draw score
    ctx.fillStyle = '#ffffff';
    ctx.font = (12 * (canvas.width/512)) + 'px monospace';
    ctx.fillText(`Score: ${state.player.score}`, 8, 18);

    // Game over text
    if (state.gameOver) {
        ctx.fillStyle = '#ffffff';
        ctx.font = (24 * (canvas.width/512)) + 'px monospace';
        ctx.fillText('Game Over!', canvas.width/4, canvas.height/2);
    }
})();

# Basic physics system
let gravity = 0.5;
let friction = 0.8;

func update_physics() {
    # Apply gravity
    player.vy = player.vy + gravity;
    
    # Apply velocity
    player.x = player.x + player.vx;
    player.y = player.y + player.vy;
    
    # Apply friction
    player.vx = player.vx * friction;
    
    # Handle collisions
    if (collide(player, ground)) {
        player.y = ground.y;
        player.vy = 0;
    }
}

Sample Games

Here's a complete example of a simple platformer game in ChipScript:

# Simple Platformer in ChipScript
setup {
    # Create player with position and physics
    let player = [
        x: 32, y: 32,      # Position
        vx: 0, vy: 0,      # Velocity
        width: 4, height: 4,
        can_jump: false
    ];
    
    # Create platforms
    let platforms = [
        [x: 10, y: 50, width: 20, height: 2],
        [x: 40, y: 40, width: 20, height: 2]
    ];
    
    # Physics constants
    let gravity = 0.5;
    let jump_force = -8;
    
    # Input handler for jumping
    on key_press(key) {
        if (key == "UP" && player.can_jump) {
            player.vy = jump_force;
            sound "jump" [volume: 0.5];
        }
    }
}

# Main game loop 
loop {
    # Apply gravity and movement
    player.vy = player.vy + gravity;
    
    if (key_down("LEFT")) {
        player.vx = -2;
    } else if (key_down("RIGHT")) {
        player.vx = 2;
    } else {
        player.vx = player.vx * 0.8;  # Friction
    }
    
    # Update position
    player.x = player.x + player.vx;
    player.y = player.y + player.vy;
    
    # Check platform collisions
    player.can_jump = false;
    for (platform in platforms) {
        if (rect_overlap(player, platform)) {
            if (player.vy > 0) {
                player.y = platform.y - player.height;
                player.vy = 0;
                player.can_jump = true;
            }
        }
    }
    
    # Keep in bounds and draw
    player.x = clamp(player.x, 0, 60);
    player.y = clamp(player.y, 0, 60);
    
    clear(#000033);
    for (platform in platforms) {
        rect(platform.x, platform.y, platform.width, platform.height, #666666);
    }
    rect(player.x, player.y, player.width, player.height, #00ff00);
}

Best Practices

  • Keep your game state organized in well-named variables
  • Use lists to manage collections of similar objects
  • Use the built-in entity manager for game objects
  • Pre-calculate values when possible
  • Use the debug display mode (3) to monitor performance
  • Batch similar draw operations together
  • Use object pooling for frequently created/destroyed objects
